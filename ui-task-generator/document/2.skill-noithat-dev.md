# Skill: noithat-dev

## Mô tả

Guideline bắt buộc tuân thủ khi viết code cho dự án **Nội Thất Tiện Lợi** — Next.js e-commerce nội thất.

Áp dụng cho mọi task: thêm feature, sửa bug, refactor, hoặc viết test.

## Mục tiêu

- Code theo đúng **3-layer architecture** (Repository → Service → Route)
- Tuân thủ convention về naming, file structure, và patterns
- Sinh code dễ review, maintain, và mở rộng
- Không sửa lung tung — chỉ thay đổi những gì cần thiết


## Pattern Templates

### Repository Pattern

```typescript
// src/server/repositories/{name}.repository.ts
import { prisma } from '@/lib/prisma';
import type { SomeInput } from '@/server/validators/{name}.validator';

const listSelect = {
  id: true,
  name: true,
  // ...chỉ lấy field cần thiết
};

export const {name}Repository = {
  async findAll(onlyActive = false) {
    const where = onlyActive ? { isActive: true } : undefined;
    return prisma.{model}.findMany({
      where,
      select: listSelect,
      orderBy: { sortOrder: 'asc' },
    });
  },

  async findById(id: string) {
    return prisma.{model}.findUnique({ where: { id } });
  },

  async findBySlug(slug: string) {
    return prisma.{model}.findUnique({ where: { slug } });
  },

  async create(data: SomeInput) {
    return prisma.{model}.create({ data });
  },

  async update(id: string, data: Partial<SomeInput>) {
    return prisma.{model}.update({ where: { id }, data });
  },

  async delete(id: string) {
    return prisma.{model}.delete({ where: { id } });
  },
};
```

### Service Pattern

```typescript
// src/server/services/{name}.service.ts
import { {name}Repository } from '@/server/repositories/{name}.repository';
import { validate{Name}, type {Name}Input } from '@/server/validators/{name}.validator';
import { createSlug } from '@/lib/utils';
import { NotFoundError, ValidationError, DuplicateError, ConflictError } from '@/server/errors';

export const {name}Service = {
  async getAll() {
    return {name}Repository.findAll(false);
  },

  async getById(id: string) {
    const result = await {name}Repository.findById(id);
    if (!result) throw new NotFoundError('Không tìm thấy');
    return result;
  },

  async create(input: Record<string, unknown>) {
    // Auto-generate slug nếu chưa có
    if (!input.slug && input.name) {
      input.slug = createSlug(input.name as string);
    }

    const validated = validate{Name}(input);
    if (!validated.success) {
      throw new ValidationError('Dữ liệu không hợp lệ', validated.error.flatten().fieldErrors);
    }

    const existing = await {name}Repository.findBySlug(validated.data.slug);
    if (existing) {
      throw new DuplicateError('Slug', validated.data.slug);
    }

    return {name}Repository.create(validated.data);
  },

  async update(id: string, input: Record<string, unknown>) {
    const current = await {name}Repository.findById(id);
    if (!current) throw new NotFoundError('Không tìm thấy');

    const validated = validate{Name}({ ...current, ...input });
    if (!validated.success) {
      throw new ValidationError('Dữ liệu không hợp lệ', validated.error.flatten().fieldErrors);
    }

    return {name}Repository.update(id, validated.data);
  },

  async delete(id: string) {
    const current = await {name}Repository.findById(id);
    if (!current) throw new NotFoundError('Không tìm thấy');
    return {name}Repository.delete(id);
  },
};
```

### Validator Pattern

```typescript
// src/server/validators/{name}.validator.ts
import { z } from 'zod';

const slugRegex = /^[a-z0-9]+(?:-[a-z0-9]+)*$/;

export const {name}Schema = z.object({
  name: z.string().min(1, 'Tên là bắt buộc').max(200),
  slug: z.string().regex(slugRegex, 'Slug không hợp lệ'),
  // ... các field khác
  sortOrder: z.coerce.number().min(0).default(0),
  isActive: z.boolean().default(true),
});

export type {Name}Input = z.infer<typeof {name}Schema>;

export function validate{Name}(data: unknown) {
  return {name}Schema.safeParse(data);
}
```

### API Route Pattern

```typescript
// src/admin/api/{name}/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { {name}Service } from '@/server/services/{name}.service';
import { parseAppError } from '@/server/errors';

// GET /admin/api/{name}
export async function GET() {
  try {
    const data = await {name}Service.getAll();
    return NextResponse.json({ success: true, data });
  } catch (error) {
    const appError = parseAppError(error);
    return NextResponse.json(
      { success: false, error: appError.message },
      { status: appError.statusCode }
    );
  }
}

// POST /admin/api/{name}
export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const data = await {name}Service.create(body);
    return NextResponse.json({ success: true, data }, { status: 201 });
  } catch (error) {
    const appError = parseAppError(error);
    const response: Record<string, unknown> = { success: false, error: appError.message };
    if (appError.details) response.errors = appError.details;
    return NextResponse.json(response, { status: appError.statusCode });
  }
}
```

### Form Client Component Pattern

```typescript
// src/admin/features/{name}/{Name}Form.tsx
'use client';

import { useState } from 'react';
import { useRouter } from 'next/navigation';
import Link from 'next/link';

interface Props {
  item?: { id: string; name: string; /* ... */ };
}

export function {Name}Form({ item }: Props) {
  const router = useRouter();
  const isEdit = !!item;
  const [loading, setLoading] = useState(false);
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [globalError, setGlobalError] = useState('');

  const [form, setForm] = useState({
    name: item?.name || '',
    // ... các field khác
  });

  function handle(e: React.ChangeEvent<HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement>) {
    const { name, value, type } = e.target;
    const v = type === 'checkbox' ? (e.target as HTMLInputElement).checked : value;
    setForm((p) => ({ ...p, [name]: v }));
    if (errors[name]) setErrors((p) => { const n = { ...p }; delete n[name]; return n; });
    setGlobalError('');
  }

  async function submit(ev: React.FormEvent) {
    ev.preventDefault();
    // Validation client-side
    const e: Record<string, string> = {};
    if (!form.name.trim()) e.name = 'Bắt buộc';
    if (Object.keys(e).length) { setErrors(e); return; }

    setLoading(true);
    setGlobalError('');
    try {
      const payload = { name: form.name.trim() /* ... */ };
      const url = isEdit ? `/admin/api/{name}/${item.id}` : '/admin/api/{name}';
      const res = await fetch(url, {
        method: isEdit ? 'PUT' : 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });
      const json = await res.json();
      if (!res.ok) {
        if (json.errors) {
          const fe: Record<string, string> = {};
          Object.entries(json.errors).forEach(([k, v]) => { fe[k] = Array.isArray(v) ? (v as string[])[0] : String(v); });
          setErrors(fe);
        } else setGlobalError(json.error || 'Lỗi');
        return;
      }
      router.push('/admin/{name}');
      router.refresh();
    } catch { setGlobalError('Lỗi kết nối'); }
    finally { setLoading(false); }
  }

  return (
    <form onSubmit={submit} noValidate>
      {globalError && <div className="alert alert-danger py-2">{globalError}</div>}
      {/* Form fields */}
      <button type="submit" disabled={loading}>
        {loading ? 'Đang lưu...' : isEdit ? 'Cập nhật' : 'Tạo mới'}
      </button>
    </form>
  );
}
```

| Tình huống | Lệnh |
|---|---|
| Thay đổi model, cần migrate DB | `npx prisma migrate dev --name <ten>` |
| Chỉ update Prisma Client types | `npx prisma generate` |
| Xem trạng thái migration | `npx prisma migrate status` |
| Reset DB (dev only) | `npx prisma migrate reset` |
| Xem DB qua UI | `npx prisma studio` |
| Push schema lên DB (không migration) | `npx prisma db push` |

**Workflow dev phổ biến:**
```bash
npx prisma generate   # Sau khi đổi schema
npx prisma db push    # Sync lên database
npm run dev           # Chạy development
```
